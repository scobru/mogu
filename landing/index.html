// Example of using Mogu
import { Mogu } from 'mogu';

const mogu = new Mogu({
    storageService: 'PINATA',
    encryption: {
        enabled: true,
        key: 'your-secret-key'
    }
});

// Backup your data
await mogu.backupFiles('./documents');
